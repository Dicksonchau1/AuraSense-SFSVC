aura_inspect.py 
from aurasense import Inspector, FacadeMap

# Initialize autonomous inspection
inspector = Inspector("building_42")
inspector.configure(
    resolution=0.2,  # mm precision
    defect_classes=["crack", "spall", "corrosion"],
    flight_mode="adaptive_spiral"
)

# Execute inspection & generate facade map
results = inspector.run(autonomous=True)
facade = FacadeMap.from_inspection(results)
facade.export("facade_report.glb", format="bim")
